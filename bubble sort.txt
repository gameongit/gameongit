A]	Write a program to implement Bubble Sort.

#include<stdio.h>
#include<malloc.h>
int arr[20];
void bubblesort(int n)
{
	int i,j,temp;
	for(i=0;i<n;i++)
	{
	for(j=0;j<n-i-1;j++)
	{
		if(arr[j]>arr[j+1])
		{
			temp=arr[j];
			arr[j]=arr[j+1];
			arr[j+1]=temp;
		}
	}
	}
}
void main()
{
	int n,i;
	printf("Enter number of elements  ");
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		printf("Enter %dth element",i);
		scanf("%d",&arr[i]);
	}
	bubblesort(n);
	for(i=0;i<n;i++)
		printf("%d\n",arr[i]);
}
