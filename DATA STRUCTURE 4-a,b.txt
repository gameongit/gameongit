a. Write a program to implement the concept of Queue with Insert, Delete, Display and Exit operations.

#include<stdio.h>
#define SIZE 6
int queue[SIZE];
int front=-1,rear=-1;
void insert();
int delete();
void display();
int main()
{
	int ch,val;
	do
	{
		printf("\n*****MAIN MENU*****\n");
		printf("1.Insert an Element\n");
		printf("2.Delete an Element\n");
		printf("3.Display\n");
		printf("4.Exit\n");
		printf("Enter your choice:   ");
		scanf("%d",&ch);
		printf("\n");
		switch(ch)
		{
			case 1:insert();
			break;
			case 2:val=delete();
			if(val!=-1)
				printf("The value deleted is:%d\n",val);
			break;
			case 3:display();
			break;
			case 4:break;
		}
	}while(ch!=4);
	return 0;
}
void display()
{
	int i;
	if(front==-1)
		printf("Queue is Empty\n");
	else
	{
		printf("The Queue is:   ");
		for(i=front;i<=rear;i++)
			printf("\t%d",queue[i]);
	}
}
void insert()
{
	int val;
	printf("Enter Value:   ");
	scanf("%d",&val);
	if(rear==SIZE-1)
		printf("Queue is Full\n");
	else if(front==-1)
	{
		front=rear=0;
		queue[rear]=val;
		printf("Successfully Inserted\n");
	}
	else
	{
		queue[++rear]=val;
		printf("Successfully Inserted\n");
	}
}
int delete()
{
	int val;
	if(front==-1)
	{
		printf("Queue is Empty\n");
		return-1;
	}
	else if(front==rear)
	{
		int val=queue[front];
		front=rear=-1;
		return val;
	}
	else
	{
		val=queue[front++];
		return val;
	}
}

B. Write a program to implement the concept of Circular Queue.

#include<stdio.h>
#define size 5
int queue[size];
int front=-1,rear=-1;
void insert();
int delete();
void display();
int main()
{
	int ch,val;
	do
	{
		printf("\n*****MAIN MENU*****\n");
		printf("1.Insert\n");
		printf("2.Delete\n");
		printf("3.Display\n");
		printf("4.Exit\n");
		printf("Enter your choice:   ");
		scanf("%d",&ch);
		printf("\n");
		switch(ch)
		{
			case 1:insert();
			break;
			case 2:val=delete();
			if(val!=-1)
				printf("The value deleted is:%d\n",val);
			break;
			case 3:display();
			break;
			case 4:break;
		}
	}while(ch!=4);
	return 0;
}
void display()
{
	int i;
	if(front==-1)
		printf("Queue is Empty\n");
	else
	{
		printf("The Queue is:   ");
		for(i=front;i!=rear;i=(i+1)%size)
			printf("\t%d",queue[i]);
		printf("\t%d",queue[i]);
	}
	printf("\n");
}
void insert()
{
	int val;
	printf("Enter Value:   ");
	scanf("%d",&val);
	if((rear+1)%size==front)
		printf("Queue is Full\n");
	else if(front==-1)
	{
		front=rear=0;
		queue[rear]=val;
		printf("Successfully Inserted\n");
	}
	else
	{
		rear=(rear+1)%size;
		queue[rear]=val;
		printf("Successfully Inserted\n");
	}
}
int delete()
{
	int val;
	if(front==-1)
	{
		printf("Queue is Empty\n");
		return-1;
	}
	else if(front!=rear)
	{
		int val=queue[front];
		front=(front+1)%size;
		return val;
	}
	else
	{
		val=queue[front];
		front=-1;
		rear=-1;
		return val;
	}
}
